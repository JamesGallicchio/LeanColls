class LeanColls.Queue (C : Type u) (τ : outParam (Type v)) :
Type (max u v)
  • model : C → List τ
  • empty : C
  • h_empty : [] = model empty
  • enq : C → τ → C
  • h_enq : ∀ (c : C) (t : τ), model c ++ [t] = model (enq c t)
  • deq : C → Option (τ × C)
  • h_deq : ∀ (c : C), List.front? (model c) = Option.map (fun x => match x with | (x, c') => (x, model c')) (deq c)
