inductive LazyList (α : Type u) :
Type u
  • nil: {α : Type u} → LazyList α
  • cons: {α : Type u} → α → LazyList α → LazyList α
  • delayed: {α : Type u} → Thunk (LazyList α) → LazyList α
source
instance LazyList.instInhabitedLazyList {α : Type u} :
Inhabited (LazyList α)
Equations
  • LazyList.instInhabitedLazyList = { default := LazyList.nil }

def LazyList.pure {α : Type u} :

LazyList α → Nat

  • LazyList.length (LazyList.cons x_1 as) = LazyList.length as + 1
