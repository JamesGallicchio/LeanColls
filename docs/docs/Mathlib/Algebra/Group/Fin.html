    theorem Fin.coe_sub_one {n : ℕ} (a : Fin (n + 1)) :
    ↑(a - 1) = if a = 0 then n else ↑a - 1

    @[simp]
    theorem Fin.lt_sub_one_iff {n : ℕ} {k : Fin (n + 2)} :
    k < k - 1 ↔ k = 0
